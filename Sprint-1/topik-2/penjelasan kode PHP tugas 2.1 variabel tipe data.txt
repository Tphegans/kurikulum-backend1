				Penjelasan Kode PHP  Tugas 2.1 Variabel Tipe data 

> var_dump($d+$e); 
	Karna nilai d=3 dan niiai e=2 maka hasilnya=5.
> var_dump(#e==$f);
	Karna tipe data e dan f sama maka booleannya=true.
> var_dump($e===$f);
	 Karna tipe data e dan f berbeda maka booleannya= false.
> var _dump($g);
	Karna nilai g= null yang artinya hampa/kosong.
>var_dump((bool)$g); 
	Karna nilai g= 0 maka booleannya= false.
>var_dump((int)$g);
	Karna nilai g=0 maka hasil integer= 0.


>var_dump(true&&false);
	Karna nilai true dan true bertipe &&/dan (logika) maka hasil booleannya= true
>var_dump(true||false);
	Karna nilai true dan false bertipe ||(atau) maka hasil booleannya true
>var_dump(false||false);
	Karna false dan false adalah sama dan bertipe || (atau) maka hasil booleannya adalah false
>var_dump(!false);
	Karna nilai false bertipe ! yang artinya (tidak)
